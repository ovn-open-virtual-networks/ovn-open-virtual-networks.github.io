How to Configure A Powerful Open vSwitch - 2021-03-18

Welcome to my blog! This post will show you a powerful way to configure openvswitch. Please check Open vSwitch for more info.

Full-NAT Example

ip netns add n1
ip netns add n2
ip link set veth1 netns n1
ip link set veth2 netns n2
ip netns exec n1 ip link set dev veth1 address 00:11:22:33:44:51
ip netns exec n2 ip link set dev veth2 address 00:11:22:33:44:61
ip netns exec n1 ip addr add 1.1.1.100/24 dev veth1
ip netns exec n2 ip addr add 2.2.2.101/24 dev veth2
ip netns exec n2 ip addr add 2.2.2.102/24 dev veth2
ip netns exec n2 ip addr add 2.2.2.103/24 dev veth2
ip netns exec n1 ip neig add 1.1.1.1 lladd 00:11:22:33:44:61 dev veth1
ip netns exec n2 ip neig add 2.2.2.1 lladd 00:11:22:33:44:51 dev veth2
ip netns exec n1 ip link set dev veth1 up
ip netns exec n2 ip link set dev veth2 up
ip netns exec n1 ip rout add default via 1.1.1.1
ip netns exec n2 ip rout add default via 2.2.2.1
$OVS_INSTALL_BIN/ovs-vsctl –may-exist add-br br-int
$OVS_INSTALL_BIN/ovs-vsctl –may-exist add-port br-int veth-n1 – set interface veth-n1 ofport_request=1
$OVS_INSTALL_BIN/ovs-vsctl –may-exist add-port br-int veth-n2 – set interface veth-n2 ofport_request=2
ip li set dev veth-n1 up
ip li set dev veth-n2 up
ip li set dev br-int up
$OVS_INSTALL_BIN/ovs-ofctl del-flows br-int
$OVS_INSTALL_BIN/ovs-ofctl add-flow br-int
“table=0,priority=100,in_port=veth-n1,ip,ip_dst=3.3.3.100,tcp action=ct(zone=100,nat(dst=2.2.2.100-2.2.2.103),commit),ct(zone=200,nat(src=3.3.3.1),commit),veth-n2”
$OVS_INSTALL_BIN/ovs-ofctl add-flow br-int
“table=0,priority=100,in_port=veth-n2,ip,ip_dst=3.3.3.1,tcp,ct_state=-trk action=ct(zone=200,nat),ct(zone=100,nat,table=0)”
$OVS_INSTALL_BIN/ovs-ofctl add-flow br-int
“table=0,priority=100,in_port=veth-n2,ip,ip_src=3.3.3.100,tcp,ct_zone=100,ct_state=+trk+est action=veth-n1”
